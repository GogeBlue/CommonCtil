97 void cmtLockEnter(cmtLock* lock)
98 {
99  cmtBool SpinState;
100 
101  if (lock->state) cmtSysLockEnter(lock->handle);
102  else
103  {
104  SpinState = cmtSpinLockEnter(&lock->value, lock->MaxSpin);
105  if (SpinState)
106  {
107  //先旋完
108  cmtSpinLockEnter(&lock->value, -1);
109  //再次检查锁状态
110  //如果已经升级了，就使用系统锁，停止使用自旋锁
111  if (lock->state)
112  {
113  cmtSysLockEnter(lock->handle);
114  cmtSpinLockLeave(lock->value);
115  }
116  //如果没有升级，就自己升级
117  else
118  {
119  //尝试升级锁
120  lock->handle = cmtSysLockInit();
121  //如果成功，切换状态，停止使用自旋锁
122  if (lock->handle)
123  {
124  lock->state = TRUE;
125  cmtSpinLockLeave(lock->value);
126  }
127  }
128  }
129  }
130 }
